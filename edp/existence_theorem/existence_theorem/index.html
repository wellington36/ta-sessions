def picard_iteration(fun, t_eval, y0, yk, args):

    pieces = [quad(func = fun, 
                   a = t_eval[i], 
                   b = t_eval[i+1], 
                   args = ((yk[i] + yk[i+1])/2, *args)
                  )[0] for i in range(len(t_eval)-1)]
    pieces.insert(0,0)
    yk = np.cumsum(np.array(pieces)) + y0

    return yk
